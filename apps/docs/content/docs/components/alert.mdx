## Props

### Alert (Root)

| Prop | Type | Default | Description |
| :--- | :--- | :--- | :--- |
| `variant` | `'info' \| 'success' \| 'warning' \| 'error'` | `'info'` | The visual style of the alert. |
| `children` | `ReactNode` | - | The content of the alert. |
| `className` | `string` | - | Additional CSS classes. |

### Alert.Icon

| Prop | Type | Default | Description |
| :--- | :--- | :--- | :--- |
| `children` | `ReactNode` | - | The icon component to display. |
| `className` | `string` | - | Additional CSS classes. |

### Alert.Title

| Prop | Type | Default | Description |
| :--- | :--- | :--- | :--- |
| `children` | `ReactNode` | - | The title of the alert. |
| `className` | `string` | - | Additional CSS classes. |

### Alert.Description

| Prop | Type | Default | Description |
| :--- | :--- | :--- | :--- |
| `children` | `ReactNode` | - | The content/description of the alert. |
| `className` | `string` | - | Additional CSS classes. |

### Alert.Dismiss

| Prop | Type | Default | Description |
| :--- | :--- | :--- | :--- |
| `onClick` | `() => void` | - | Callback fired when the dismiss button is clicked. |
| `children` | `ReactNode` | - | Optional custom content for the dismiss button. |
| `className` | `string` | - | Additional CSS classes. |

## Source Code

```tsx title="src/components/ui/alert.tsx"
import React, { HTMLAttributes, ButtonHTMLAttributes, forwardRef } from "react";
import { cn } from "@/lib/cn";

// ============================================================================
// Types & Interfaces
// ============================================================================

type AlertVariant = "info" | "success" | "warning" | "error";

interface AlertRootProps extends HTMLAttributes<HTMLDivElement> {
    variant?: AlertVariant;
    dismissible?: boolean;
    onDismiss?: () => void;
    children: React.ReactNode;
}

// ============================================================================
// Styling Maps
// ============================================================================

const variantStyles: Record<AlertVariant, string> = {
    info: "bg-info-50 border-info-200 text-info-900 dark:bg-info-950 dark:border-info-800 dark:text-info-100",
    success: "bg-success-50 border-success-200 text-success-900 dark:bg-success-950 dark:border-success-800 dark:text-success-100",
    warning: "bg-warning-50 border-warning-200 text-warning-900 dark:bg-warning-950 dark:border-warning-800 dark:text-warning-100",
    error: "bg-error-50 border-error-200 text-error-900 dark:bg-error-950 dark:border-error-800 dark:text-error-100",
};

// Maps for ARIA roles and live regions to ensure WCAG 2.2 AA compliance
const variantRole: Record<AlertVariant, "alert" | "status"> = {
    info: "status",
    success: "status",
    warning: "alert",
    error: "alert",
};

// ============================================================================
// Alert Root Component (Server-safe, no state!)
// ============================================================================

const AlertRoot = forwardRef<HTMLDivElement, AlertRootProps>(({
    variant = "info",
    dismissible = false,
    className = "",
    children,
    ...props
}, ref) => {
    return (
        <div
            ref={ref}
            role={variantRole[variant]}
            className={cn(
                "relative flex gap-3 p-4 border rounded-lg w-full",
                variantStyles[variant],
                className
            )}
            data-variant={variant}
            {...props}
        >
            {children}
        </div>
    );
});
AlertRoot.displayName = "Alert";

// ============================================================================
// Alert Icon Component
// ============================================================================

const AlertIcon = forwardRef<HTMLDivElement, HTMLAttributes<HTMLDivElement>>(({
    className = "",
    children,
    ...props
}, ref) => {
    return (
        <div
            ref={ref}
            className={cn("shrink-0", className)}
            aria-hidden="true" // Icons are decorative in this context
            {...props}
        >
            {children}
        </div>
    );
});
AlertIcon.displayName = "Alert.Icon";

// ============================================================================
// Alert Title Component
// ============================================================================

const AlertTitle = forwardRef<HTMLHeadingElement, HTMLAttributes<HTMLHeadingElement>>(({
    className = "",
    children,
    ...props
}, ref) => {
    return (
        <h5
            ref={ref}
            className={cn("font-semibold mb-1 leading-none tracking-tight", className)}
            {...props}
        >
            {children}
        </h5>
    );
});
AlertTitle.displayName = "Alert.Title";

// ============================================================================
// Alert Description Component
// ============================================================================

const AlertDescription = forwardRef<HTMLDivElement, HTMLAttributes<HTMLDivElement>>(({
    className = "",
    children,
    ...props
}, ref) => {
    return (
        <div
            ref={ref}
            className={cn("text-sm [&_p]:leading-relaxed", className)}
            {...props}
        >
            {children}
        </div>
    );
});
AlertDescription.displayName = "Alert.Description";

// ============================================================================
// Alert Dismiss Component
// ============================================================================

interface AlertDismissProps extends ButtonHTMLAttributes<HTMLButtonElement> {
    onDismiss?: () => void;
}

const AlertDismiss = forwardRef<HTMLButtonElement, AlertDismissProps>(({
    className = "",
    children,
    onClick,
    ...props
}, ref) => {
    const defaultIcon = (
        <svg
            xmlns="http://www.w3.org/2000/svg"
            viewBox="0 0 24 24"
            fill="none"
            stroke="currentColor"
            strokeWidth="2"
            strokeLinecap="round"
            strokeLinejoin="round"
            className="w-4 h-4"
        >
            <line x1="18" y1="6" x2="6" y2="18" />
            <line x1="6" y1="6" x2="18" y2="18" />
        </svg>
    );

    return (
        <button
            ref={ref}
            type="button"
            onClick={onClick}
            className={cn(
                "absolute top-4 right-4 rounded-md p-1 transition-colors hover:bg-black/10 dark:hover:bg-white/10 focus-visible:outline-none focus-visible:ring-2 ring-offset-2",
                className
            )}
            aria-label="Dismiss alert"
            {...props}
        >
            {children || defaultIcon}
        </button>
    );
});
AlertDismiss.displayName = "Alert.Dismiss";

// ============================================================================
// Content Wrapper Component
// ============================================================================

const AlertContent = forwardRef<HTMLDivElement, HTMLAttributes<HTMLDivElement>>(({
    className = "",
    children,
    ...props
}, ref) => {
    return (
        <div
            ref={ref}
            className={cn("flex-1", className)}
            {...props}
        >
            {children}
        </div>
    );
});
AlertContent.displayName = "Alert.Content";

// ============================================================================
// Compound Component Export
// ============================================================================

export const Alert = Object.assign(AlertRoot, {
    Icon: AlertIcon,
    Title: AlertTitle,
    Description: AlertDescription,
    Dismiss: AlertDismiss,
    Content: AlertContent,
});

export default Alert;
```
