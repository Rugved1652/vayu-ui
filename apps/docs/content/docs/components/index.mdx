export const Badge = ({ children, variant }) => {
  const colors = {
    server:
      'bg-green-100 text-green-800 dark:bg-green-900/30 dark:text-green-300 border-green-200 dark:border-green-800',
    client:
      'bg-blue-100 text-blue-800 dark:bg-blue-900/30 dark:text-blue-300 border-blue-200 dark:border-blue-800',
  };
  return (
    <span
      className={`text-[10px] px-2 py-0.5 rounded-full border font-medium uppercase tracking-wide ${colors[variant]}`}
    >
      {children}
    </span>
  );
};

export const CardTitle = ({ children, type }) => (
  <div className="flex items-center gap-2">
    <span>{children}</span>
    <Badge variant={type}>{type}</Badge>
  </div>
);
