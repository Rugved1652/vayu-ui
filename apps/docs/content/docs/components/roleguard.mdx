    ```tsx
    import { RoleProvider, RoleGuard, RequireAuth, useRoles } from "@/components/ui/roleguard"

    // 1. Wrap your app with RoleProvider
    <RoleProvider roles={["admin"]} permissions={["orders:write"]}>
      <App />
    </RoleProvider>

    // 2. Guard any section
    <RoleGuard allowed={["admin"]} fallback={<p>Access denied</p>}>
      <AdminPanel />
    </RoleGuard>

    // 3. Redirect on deny
    <RoleGuard allowed={["user"]} redirectTo="/login">
      <Dashboard />
    </RoleGuard>

    // 4. Permission-gated
    <RoleGuard permissions={["orders:write"]} fallback={<p>Read-only</p>}>
      <OrderActions />
    </RoleGuard>

    // 5. Deny-list (block specific roles)
    <RoleGuard denied={["guest"]} fallback={<p>Members only</p>}>
      <Content />
    </RoleGuard>

    // 6. RequireAuth (any role = authenticated)
    <RequireAuth redirectTo="/login">
      <ProtectedPage />
    </RequireAuth>

    // 7. useRoles hook
    const { hasRole, hasPermission } = useRoles();
    ```
  </Tab>

## Architecture

```
RoleProvider (context)
  └── RoleGuard (conditional render / redirect)
        ├── allowed? → match "some" or "every"
        ├── denied?  → deny-list takes priority
        ├── permissions? → all must match
        └── redirectTo? → client-side redirect
```

## Components

| Component | Description |
|:----------|:------------|
| `RoleProvider` | Context provider — wraps app with `roles` + `permissions` |
| `RoleGuard` | Conditional renderer — gates children on role/permission checks |
| `RequireAuth` | Convenience wrapper — ensures user has at least one role |

## RoleGuard Props

| Prop | Type | Default | Description |
|:-----|:-----|:--------|:------------|
| `allowed` | `string[]` | — | Roles that grant access |
| `denied` | `string[]` | — | Roles that explicitly block access |
| `permissions` | `string[]` | — | All listed permissions must be present |
| `match` | `'some' \| 'every'` | `'some'` | How `allowed` roles are matched |
| `redirectTo` | `string` | — | URL to redirect on denial |
| `fallback` | `ReactNode` | `null` | Shown when access is denied |
| `loading` | `ReactNode` | `null` | Shown while resolving |

## useRoles Hook

```tsx
const {
  roles,           // string[]
  permissions,     // string[]
  hasRole,         // (role) => boolean
  hasAnyRole,      // (roles[]) => boolean
  hasEveryRole,    // (roles[]) => boolean
  hasPermission,   // (perm) => boolean
  hasAllPermissions // (perms[]) => boolean
} = useRoles();
```
