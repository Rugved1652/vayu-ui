## Features

- Three sizes (`sm`, `md`, `lg`) for different contexts
- Built with design system tokens for consistent styling
- WCAG 2.2 AA compliant with `role="status"`, `aria-live`, and `aria-busy`
- Smooth CSS animation with Tailwind's `animate-spin`
- Respects `prefers-reduced-motion` for accessibility
- Customizable colors via className override
- Dark mode support with semantic color tokens
- Server-side rendering compatible

## Anatomy

```tsx
<Spinner
  size="md"
  aria-label="Loading content"
/>
```

## Props

Extends `HTMLAttributes<HTMLSpanElement>`.

| Prop | Type | Default | Description |
| :--- | :--- | :------ | :---------- |
| `size` | `"sm" \| "md" \| "lg"` | `"md"` | Size of the spinner |
| `aria-label` | `string` | `"Loading"` | Accessible label for screen readers |
| `className` | `string` | — | Additional CSS classes |

## Sizes

| Size | Dimensions | Use Case |
| :--- | :--------- | :------- |
| `sm` | 16×16 px | Inline text, buttons, small indicators |
| `md` | 24×24 px | Default, general loading states |
| `lg` | 32×32 px | Full-page loaders, prominent areas |

## Accessibility

This component follows WCAG 2.2 AA guidelines:

- **Semantic HTML**: Uses `<span>` with `role="status"` for live regions
- **Live Announcements**: `aria-live="polite"` announces loading state to screen readers
- **Busy State**: `aria-busy="true"` indicates the element is in a loading state
- **Screen Readers**: Includes visually hidden text for screen reader users
- **ARIA Labels**: Supports custom `aria-label` for context
- **Reduced Motion**: Animation respects `prefers-reduced-motion`

### Screen Reader Behavior

The spinner announces "Loading..." to screen readers when it appears. Use a descriptive `aria-label` for context:

```tsx
<Spinner aria-label="Loading search results" />
```

## Source Code

```tsx title="packages/ui/src/components/ui/spinner.tsx"
import { HTMLAttributes } from "react";
import { cn } from "./utils";

// ============================================================================
// Types
// ============================================================================

type SpinnerSize = "sm" | "md" | "lg";

interface SpinnerProps extends HTMLAttributes<HTMLSpanElement> {
    /** Size of the spinner */
    size?: SpinnerSize;
    /** Accessible label for screen readers */
    "aria-label"?: string;
}

// ============================================================================
// Size Configuration
// ============================================================================

const sizeClasses: Record<SpinnerSize, string> = {
    sm: "w-4 h-4",
    md: "w-6 h-6",
    lg: "w-8 h-8",
};

// ============================================================================
// Component
// ============================================================================

/**
 * Spinner — A WCAG 2.2 AA compliant loading indicator component.
 *
 * Displays an animated circular spinner using design system tokens.
 * Fully accessible with proper ARIA attributes for screen readers.
 *
 * @example
 * ```tsx
 * <Spinner size="md" aria-label="Loading content" />
 * ```
 */
function Spinner({
    className,
    size = "md",
    "aria-label": ariaLabel = "Loading",
    ...props
}: SpinnerProps) {
    return (
        <span
            role="status"
            aria-live="polite"
            aria-busy="true"
            aria-label={ariaLabel}
            className={cn(
                "rounded-full border-2 border-t-transparent",
                "animate-spin motion-reduce:animate-none",
                "border-primary-500 dark:border-primary-400 dark:border-t-transparent",
                sizeClasses[size],
                className
            )}
            {...props}
        >
            <span className="sr-only">{ariaLabel}</span>
        </span>
    );
}

export { Spinner };
export type { SpinnerProps, SpinnerSize };
```
