## Props

### TextArea.Root

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `variant` | `"default" \| "outline"` | `"default"` | Visual variant |
| `size` | `"sm" \| "md" \| "lg"` | `"md"` | Size of the text area |
| `error` | `boolean` | `false` | Whether to show error state style |
| `maxLength` | `number` | - | Maximum number of characters allowed (for shared context) |
| `disabled` | `boolean` | `false` | Whether the text area is disabled |
| `className` | `string` | - | Additional class names |
| `children` | `ReactNode` | - | Component children |

### TextArea.Label

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `showCharCount` | `boolean` | `false` | Whether to show character count next to label |
| `children` | `ReactNode` | - | Label content |

### TextArea.Input

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `resize` | `"none" \| "vertical" \| "horizontal" \| "both"` | `"vertical"` | Resize behavior |
| `rows` | `number` | `4` | Number of visible text lines |
| `value` | `string \| number \| readonly string[]` | - | Controlled value |
| `onChange` | `(event: React.ChangeEvent<HTMLTextAreaElement>) => void` | - | Change event handler |
| `...props` | `TextareaHTMLAttributes` | - | Standard HTML textarea attributes |

### TextArea.SupportText

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `string \| string[]` | - | Helper text or list of helper texts |

### TextArea.ErrorText

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `string \| string[]` | - | Error message or list of error messages |

### TextArea.CharCount

Displays character count based on context. No props.

## Source Code

```tsx title="src/components/ui/textarea.tsx"
"use client";
import { AlertCircle } from "lucide-react";
import React, {
    TextareaHTMLAttributes,
    useState,
    forwardRef,
    createContext,
    useContext,
} from "react";

export type TextAreaResize = "none" | "vertical" | "horizontal" | "both";
export type TextAreaVariant = "default" | "outline";
export type TextAreaSize = "sm" | "md" | "lg";

// Context for sharing state between compound components
interface TextAreaContextValue {
    isFocused: boolean;
    setIsFocused: (focused: boolean) => void;
    charCount: number;
    setCharCount: (count: number) => void;
    maxLength?: number;
    error?: boolean;
    variant: TextAreaVariant;
    size: TextAreaSize;
    disabled?: boolean;
}

const TextAreaContext = createContext<TextAreaContextValue | undefined>(
    undefined
);

const useTextAreaContext = () => {
    const context = useContext(TextAreaContext);
    if (!context) {
        throw new Error(
            "TextArea compound components must be used within TextArea.Root"
        );
    }
    return context;
};

// Root Component
interface TextAreaRootProps {
    children: React.ReactNode;
    variant?: TextAreaVariant;
    size?: TextAreaSize;
    error?: boolean;
    maxLength?: number;
    disabled?: boolean;
    className?: string;
}

const TextAreaRoot = ({
    children,
    variant = "default",
    size = "md",
    error = false,
    maxLength,
    disabled = false,
    className = "",
}: TextAreaRootProps) => {
    const [isFocused, setIsFocused] = useState(false);
    const [charCount, setCharCount] = useState(0);

    return (
        <TextAreaContext.Provider
            value={{
                isFocused,
                setIsFocused,
                charCount,
                setCharCount,
                maxLength,
                error,
                variant,
                size,
                disabled,
            }}
        >
            <div className={`w-full flex flex-col gap-1 ${className}`}>
                {children}
            </div>
        </TextAreaContext.Provider>
    );
};

// Label Component
interface TextAreaLabelProps {
    children: React.ReactNode;
    showCharCount?: boolean;
}

const TextAreaLabel = ({
    children,
    showCharCount = false,
}: TextAreaLabelProps) => {
    const { charCount, maxLength } = useTextAreaContext();

    return (
        <div className="flex items-center justify-between px-2">
            <label className="font-primary text-neutral-900 dark:text-white text-sm font-medium">
                {children}
            </label>
            {showCharCount && (
                <span className="text-xs font-secondary text-neutral-500 dark:text-neutral-400">
                    {charCount}
                    {maxLength && `/${maxLength}`}
                </span>
            )}
        </div>
    );
};

// Input Component
interface TextAreaInputProps
    extends Omit<TextareaHTMLAttributes<HTMLTextAreaElement>, "size"> {
    resize?: TextAreaResize;
}

const TextAreaInput = forwardRef<HTMLTextAreaElement, TextAreaInputProps>(
    (
        {
            resize = "vertical",
            rows = 4,
            value,
            onChange,
            className = "",
            ...restProps
        },
        ref
    ) => {
        const {
            isFocused,
            setIsFocused,
            setCharCount,
            maxLength,
            error,
            variant,
            size,
            disabled,
        } = useTextAreaContext();

        const handleChange = (e: React.ChangeEvent<HTMLTextAreaElement>) => {
            setCharCount(e.target.value.length);
            if (onChange) {
                onChange(e);
            }
        };

        const getResizeClass = () => {
            switch (resize) {
                case "none":
                    return "resize-none";
                case "vertical":
                    return "resize-y";
                case "horizontal":
                    return "resize-x";
                case "both":
                    return "resize";
                default:
                    return "resize-y";
            }
        };

        const sizeConfig = {
            sm: {
                wrapper: "px-2.5 py-1.5",
                text: "text-sm",
            },
            md: {
                wrapper: "px-3 py-2.5",
                text: "text-base",
            },
            lg: {
                wrapper: "px-4 py-3",
                text: "text-lg",
            },
        };

        const config = sizeConfig[size];

        const variantStyles = {
            default: {
                wrapper: `bg-white dark:bg-neutral-900 border-2 ${error
                        ? "border-error-500 dark:border-error-600"
                        : isFocused
                            ? "border-primary-500 dark:border-primary-600"
                            : "border-neutral-300 dark:border-neutral-700"
                    } hover:border-primary-400 dark:hover:border-primary-600`,
            },
            outline: {
                wrapper: `bg-transparent border-2 ${error
                        ? "border-error-500 dark:border-error-600"
                        : isFocused
                            ? "border-primary-500 dark:border-primary-600"
                            : "border-neutral-300 dark:border-neutral-700"
                    } hover:border-primary-400 dark:hover:border-primary-600`,
            },
        };

        const styles = variantStyles[variant];

        return (
            <div
                className={`${styles.wrapper} ${config.wrapper} flex gap-3 w-full rounded-lg outline-0 transition-colors ${disabled ? "opacity-50 cursor-not-allowed" : ""
                    }`}
            >
                <textarea
                    ref={ref}
                    className={`w-full bg-transparent outline-0 font-secondary text-neutral-900 dark:text-white placeholder:text-neutral-500 ${config.text
                        } ${getResizeClass()} ${disabled ? "cursor-not-allowed" : ""
                        } ${className}`}
                    rows={rows}
                    value={value}
                    onChange={handleChange}
                    onFocus={() => setIsFocused(true)}
                    onBlur={() => setIsFocused(false)}
                    maxLength={maxLength}
                    disabled={disabled}
                    {...restProps}
                />
            </div>
        );
    }
);

TextAreaInput.displayName = "TextAreaInput";

// Support Text Component
interface TextAreaSupportTextProps {
    children: string | string[];
}

const TextAreaSupportText = ({ children }: TextAreaSupportTextProps) => {
    if (typeof children === "string") {
        return (
            <p className="text-xs font-secondary text-neutral-600 dark:text-neutral-400 px-2">
                {children}
            </p>
        );
    }

    return (
        <ul className="text-xs font-secondary text-neutral-600 dark:text-neutral-400 px-2 list-disc list-inside space-y-1">
            {children.map((text, idx) => (
                <li key={idx}>{text}</li>
            ))}
        </ul>
    );
};

// Error Text Component
interface TextAreaErrorTextProps {
    children: string | string[];
}

const TextAreaErrorText = ({ children }: TextAreaErrorTextProps) => {
    if (typeof children === "string") {
        return (
            <p className="text-xs font-secondary text-error-600 dark:text-error-400 px-2 flex items-center gap-1">
                <AlertCircle className="w-3 h-3" />
                {children}
            </p>
        );
    }

    return (
        <ul className="text-xs font-secondary text-error-600 dark:text-error-400 px-2 list-disc list-inside space-y-1">
            {children.map((text, idx) => (
                <li key={idx}>{text}</li>
            ))}
        </ul>
    );
};

// Character Count Component (standalone)
const TextAreaCharCount = () => {
    const { charCount, maxLength } = useTextAreaContext();

    return (
        <span className="text-xs font-secondary text-neutral-500 dark:text-neutral-400 px-2">
            {charCount}
            {maxLength && `/${maxLength}`}
        </span>
    );
};

// Export compound component
export const TextArea = {
    Root: TextAreaRoot,
    Label: TextAreaLabel,
    Input: TextAreaInput,
    SupportText: TextAreaSupportText,
    ErrorText: TextAreaErrorText,
    CharCount: TextAreaCharCount,
};
```
